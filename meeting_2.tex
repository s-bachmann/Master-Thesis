    \begin{itemize}
        \item Writing introduction
        \item Implementing feedback for introdcution
        \item Setup of remote machine
        \item Running simulations with lower source frequency
        \item Plotting waterfall plots, wiggle plots, single traces and frequency content 
    \end{itemize}

\begin{frame}{Simulation parameters used}
    \begin{itemize}
        \item 30m by 3m domain
        \item one simulation with 150m by 3m domain
        \item 3 layers, all layers 1m thick
        \item absorbing boundaries at edges of domain
        \item 10Hz vector point source 
        \item moment tensor source for skier (myy=0, mxx=0, mxy=3e4)
    \end{itemize}
    
\end{frame}
