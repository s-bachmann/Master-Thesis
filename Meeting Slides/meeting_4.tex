\begin{frame}{Overview of literature review}
    \begin{itemize}
        \item Literature review of 13 papers
        \item Main topics: crack propagation in snow, snow modeling, modeling in more general sense (introduction to different modeling methods used)
        \item Modeling methods: Material Point Method (MPM) and Discrete Element Method (DEM)
    \end{itemize}
\end{frame}

\begin{frame}{Motivation}
    \begin{itemize}
        \item Understanding the different modeling methods and their limitation is important for thesis to understand which methods are suited for what
        \item Understanding how cracks propagate and which factors influence them is important for risk assessment
        \item Understanding which factors lead to supershear crack propagation is also important for risk assessment
    \end{itemize}
\end{frame}

\begin{frame}{Modeling Methods}
    \begin{itemize}
        \item Two main methods used: MPM and DEM
        \item DEM described by \cite{cundallDiscreteNumericalModel1979}, material of disks \& spheres, contact forces and resulting displacements calculated
        \item MPM where material is discretized in points with Eulerian background mesh to calculate stresses at points and time-dependent equations, for example in \cite{gaumeDynamicAnticrackPropagation2018}
        \item MPM is more favorable for slope-scale simulations because of modelling of grain contacts
        \item Other models for snow such as SNOWPACK \parencite{barteltPhysicalSNOWPACKModel2002} and CROCUS \parencite{brunNumericalModelSimulate1992} model snowpack response to external factors
    \end{itemize}
    
\end{frame}


\begin{frame}{Avalanche Release - General Crack Propagation}
    \begin{itemize}
        \item All papers agree on avalanche formation by loading - weak layer failure
        \item Some papers discuss anticrack propagation (closure of cracks formed) like eg \cite{gaumeDynamicAnticrackPropagation2018}
        \item Others state shear cracks to be the main failure mode
        \item \cite{bergfeldSignaturesSubRayleighSupershear2025} combine the two main hypotheses: anticrack early in avalanche propagation, shearing later 
        \item Critical crack length: length of the crack needed for breakoff \parencite{gaumeSnowFractureRelation2017a}
    \end{itemize}
\end{frame}

\begin{frame}{Factors influencing crack propagation}
    \begin{itemize}
        \item Slope angle: higher slope angle generally increases crack speed and can trigger transition from sub-Rayleigh to supershear; it can also reduce the critical crack length needed for release \parencite{bobillierNumericalInvestigationCrack2024, trottetTransitionSubRayleighAnticrack2022, gaumeSnowFractureRelation2017a}
        \item Slab stiffness (elastic modulus): stiffer slabs lead to lower normalized crack speed in some DEM results, but it tends to increase critical crack length (more crack growth needed before release) \parencite{bobillierNumericalInvestigationCrack2024, gaumeSnowFractureRelation2017a}
        \item Weak layer strength \& elastic modulus: higher weak-layer elastic modulus can increase crack speed; stronger weak layers generally require more energy/longer cracks for release \parencite{bobillierNumericalInvestigationCrack2024, gaumeSnowFractureRelation2017a, gaumeModelingCrackPropagation2015}
    
    \end{itemize}
\end{frame}

\begin{frame}{Factors influencing crack propagation (continued)}
    \begin{itemize}
        \item Slab density: higher slab density is associated with faster propagation in some models, while also reducing critical crack length for release \parencite{gaumeModelingCrackPropagation2015, gaumeSnowFractureRelation2017a}
        \item Slab thickness: thicker slabs can increase propagation speed in DEM studies; in critical-crack formulations, greater slab thickness lowers critical crack length \parencite{gaumeModelingCrackPropagation2015, gaumeSnowFractureRelation2017a}, but thicker slabs can also distribute load more easily \parencite{rosendahlModelingSnowSlab2020}
        \item Weak layer thickness: thinner weak layers can lead to faster crack propagation and lower critical crack length for release \parencite{gaumeModelingCrackPropagation2015, gaumeSnowFractureRelation2017a}
        \item Snow structure and loading rate: sintering and lower loading rates can lead to more stable snowpacks, which means higher critical crack lenths \parencite{mulakNumericalInvestigationMixedmode2019}
        \item Loading rate: slow loading (like from snow accumulation) can lead to formation of new bonds (sintering): strengthens snowpack \parencite{mulakNumericalInvestigationMixedmode2019}
    \end{itemize}
    
\end{frame}


\begin{frame}{Deformation stages}
    \begin{itemize}
        \item After loading, multiple stages of deformation
        \item loading response by an elastic regime, a drop in pressure and shear stress, volumetric collapse of the weak layer and lastly dense packing \parencite{gaumeDynamicAnticrackPropagation2018}
        \item Slab bending induced by weak layer collapse at bottom of snowpack \parencite{bergfeldSignaturesSubRayleighSupershear2025}
        \item Deformation because of slab tension at top of snow \parencite{bergfeldSignaturesSubRayleighSupershear2025}

    \end{itemize}
    
\end{frame}


\begin{frame}{Crack propagation direction \& mode}
\begin{itemize}
    \item Direction of crack propagation dependent on simulation scale \parencite{trottetTransitionSubRayleighAnticrack2022}
    \item Small-scale: propagation upslope (remote triggering), because collapse depth for bending limited by weak layer collapse \parencite{trottetTransitionSubRayleighAnticrack2022}
    \item Large-scale: downslope, because propagation speed can increase because of slab tension \parencite{trottetTransitionSubRayleighAnticrack2022}
    \item Mode I (normal to crack face), Mode II (tangential to existing faces) dependent on slope angle \parencite{rosendahlModelingSnowSlab2020}
    \item In steep slopes, more mode II \parencite{rosendahlModelingSnowSlab2020}
\end{itemize}
\end{frame}


\begin{frame}[allowframebreaks]{Supershear crack propagation}
    \begin{itemize}
        \item Supershear crack propagation when cracks propagate above elastic wave speed in snow \parencite{trottetTransitionSubRayleighAnticrack2022}
        \item Snow properties influence if supershear speed is reached
        \item High slab tensile strength: supershear cracks \parencite{trottetTransitionSubRayleighAnticrack2022}
        \item Increased slab tension in steep terrains: high speeds (possible supershear) \parencite{trottetTransitionSubRayleighAnticrack2022}
        \item Slope angle dependance: at below 30°, sub-rayleigh, at above 30°: supershear \parencite{bobillierNumericalInvestigationCrack2024}
        \item Also property dependent (weak layer elastic modulus & snow elastic modulus) \parencite{bobillierNumericalInvestigationCrack2024}
        \item Generally: the "longer" cracks can propagate before breakoff, the higher the speed they can reach (direct failure in weak slabs, no failure in very strong slabs)
        \item Early stages of crack propagation are anticrack, later stages supershear \parencite{bergfeldSignaturesSubRayleighSupershear2025}
    \end{itemize}
    
\end{frame}
